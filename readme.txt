 Примеры заполнения КД.


 1. Авторасчет КД (исходя из значений аллодовской библиотеки spellLib)

 ['BARD'] = {
    ['Резонансный барьер'] = true,
 }


 2. Общее значение, независимое от ранга умения:

 ['BARD'] = {
    ['Резонансный барьер'] = 35,
 }

 ИЛИ

 ['BARD'] = {
    ['Резонансный барьер'] = {
        value = 35, -- такой подход может понадобиться, если помимо значения кд планируются доп коэффициенты (дальше, параметр "calculate")
    },
 }


 3. Значения по рангам

 ['BARD'] = {
    ['Резонансный барьер'] = {
        rank_1 = 50,
        rank_2 = 40,
        rank_3 = 35,
    },
 }


 4. Применение бонусов от вех

 ['BARD'] = {
    ['Резонансный барьер'] = {
        calculate = 'n * 0.86' -- Веха "акустическая стена" дает -14% к кд, но т.к. эта веха только для Аспекта поддержки, см дальше...
    },
 }


 5. Переопределение по аспектам.

 ['BARD'] = {
    ['Резонансный барьер'] = {
        ['Аспект Поддержки'] = {
            calculate = 'n * 0.86' -- где n - исходное значение кд, т.е. будет выполнено n = n * 0.86
        },
    },
 }


 6. Переопределение по аспектам.

  ['BARD'] = {
    ['Резонансный барьер'] = {
        ['Аспект Поддержки'] = 31, -- аналогично, можно расписать по рангам (rank_1 - rank_3), или параметром "value", использовать "calculate".
    },                             -- Синтаксис описания умения для аспекта такой же, как и без аспекта. Но приоритет, конечно, у аспекта.
  }
